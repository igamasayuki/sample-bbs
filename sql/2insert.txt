--コメントを1万件作成(記事idは1~15000でランダム)
INSERT INTO comments(name, content, article_id)
SELECT format('コメントユーザー%s',i),format('コメント内容%s',i), (random() * 15000)::int + 1
FROM generate_series(1,10000) AS i;
